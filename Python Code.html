def clean(titanic_full_set):  
    #Some records have Null value in Age column. Setting these to meadian value of the entire dataset
    titanic_full_set.loc[pandas.isnull(titanic_full_set['Age']),'Age'] = titanic_full_set['Age'].median()
    #The Male values were classified as 0, and the Female values as 1
    titanic_full_set.loc[titanic_full_set['Sex'] == 'male','Sex'] = 0
    titanic_full_set.loc[titanic_full_set['Sex'] == 'female','Sex'] = 1
    print (titanic_full_set.describe())
    return titanic_full_set

def percent_survived(titanic_full_set):
    # This function addresses the following question
    #1) WHat percentage of males and females survived?
    
    total_survived = titanic_full_set['Survived'].sum()
    male_survived = titanic_full_set.loc[titanic_full_set['Sex'] == 0,'Survived'].sum()
    total_male = titanic_full_set.loc[titanic_full_set['Sex'] == 0,'Sex'].count()

    female_survived = titanic_full_set.loc[titanic_full_set['Sex'] == 1,'Survived'].sum()
    total_female = titanic_full_set.loc[titanic_full_set['Sex'] == 1,'Sex'].count()

    #print male_survived, total_male, female_survived, total_female, total_survived
    percent_male_survived = float(male_survived)/total_male
    percent_female_survived = float(female_survived)/total_female
    return (percent_male_survived, percent_female_survived)

def pclass_Percent_Females_Survived(pclass):
    #Calculates % of females survived in pClass
    return float(pclass.loc[pclass['Sex'] == 1, 'Survived'].sum()) / (pclass['Survived'].sum())


def percent_Survived(pclass):
    #Calculates % of people survived in pClass
    return float(pclass['Survived'].sum()) / len(pclass['Survived'])
